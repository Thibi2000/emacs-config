** Inbetween parens
   #+BEGIN_SRC emacs-lisp
(defun inbetween-parentheses (beg end)
  (interactive (list (point) (mark)))
  (kill-region beg end)
  (insert "\(")
  (yank)
  (insert "\)"))
(global-set-key (kbd "C-x C-p") 'inbetween-parentheses)
   #+END_SRC
